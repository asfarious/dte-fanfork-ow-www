# Ruler in red, lean on vassals
character_event = {
	id = 38010
	desc = EVTDESC38010
	picture = GFX_evt_council

	only_playable = yes
	prisoner = no

	trigger = {
		wealth < -50

		any_vassal = {
			opinion = {
				who = ROOT
				value >= 0
			}

			wealth >= 10
			NOT = { has_character_flag = took_extra_tax }
		}
	}

	mean_time_to_happen = {
		months = 60

		mult_modifier = {
			factor = 0.25
			has_character_flag = loan_taken
		}
	}

	option = {
		name = EVTOPTA38010 # Ask one vassal for large

		trigger = {
			any_vassal = {
				wealth >= 200

				opinion = {
					who = ROOT
					value >= 0
				}

				NOT = { has_character_flag = took_extra_tax }
			}
		}

		ai_chance = {
			factor = 10
		}

		random_vassal = {
			show_scope_change = no

			limit = {
				wealth >= 200

				opinion = {
					who = ROOT
					value >= 0
				}

				NOT = { has_character_flag = took_extra_tax }
			}

			opinion = {
				name = got_money_request
				who = ROOT
				years = 2
			}

			hidden_effect = {
				set_variable = {
					which = loan_size
					value = -250
				}

				ROOT = {
					set_variable = {
						which = loan_size
						value = 250
					}
				}
			}

			set_character_flag = took_extra_tax

			character_event = {
				id = 38011
				days = 1
				tooltip = EVTTOOLTIP38013
			}
		}
	}

	option = {
		name = EVTOPTC38010 # ask all vassals for medium

		trigger = {
			any_vassal = {
				opinion = {
					who = ROOT
					value = 0
				}

				wealth >= 50
				NOT = { has_character_flag = took_extra_tax }
			}
		}

		ai_chance = {
			factor = 1

			mult_modifier = {
				factor = 10
				wealth >= -100
			}
		}

		random_vassal = {
			show_scope_change = no

			limit = {
				wealth >= 25
				NOT = { has_character_flag = took_extra_tax }
			}

			opinion = {
				name = got_money_request
				who = ROOT
				years = 2
			}

			hidden_effect = {
				set_variable = {
					which = loan_size
					value = -50
				}

				ROOT = {
					set_variable = {
						which = loan_size
						value = 50
					}
				}
			}

			set_character_flag = took_extra_tax

			character_event = {
				id = 38011
				days = 1
				tooltip = EVTTOOLTIP38012
			}
		}
	}

	option = {
		name = EVTOPTB38010 # ask all vassal for small

		trigger = {
			any_vassal = {
				opinion = {
					who = ROOT
					value = 0
				}

				wealth >= 10
				NOT = { has_character_flag = took_extra_tax }
			}
		}

		ai_chance = {
			factor = 1
		}

			show_scope_change = no

			has_character_flag = stolen_church_properties
		}
	}

	mean_time_to_happen = {
		months = 48
	}

	option = {
		name = EVTOPTA38020 # There are fat bishop, seize them !

		trigger = {
			any_vassal = {
				is_theocracy = yes

				any_demesne_title = {
					tier = BARON
					num_of_buildings >= 1
				}
			}
		}

		ai_chance = {
			factor = 1

			trigger = { trait = cynical }
		}

		random_vassal = {
			show_scope_change = no

			limit = {
				is_theocracy = yes

				any_demesne_title = {
					tier = BARON
					num_of_buildings >= 1
				}
			}

			character_event = {
				id = 38021
				days = 1
				tooltip = EVTTOOLTIP38021
			}

			imprison = ROOT
		}

		wealth = 300
		add_trait_silently_cruel_effect = yes
		set_character_flag = stolen_church_properties
	}

	option = {
		name = EVTOPTB38020 # Time are harsh, force them to share the costs !

		wealth = 100

		random = {
			chance = 25
			random_vassal = {

			}

	}

		name = EVTOPTC38020 # Compromise with them to find money !

# Offer to buy a title from indebted liege
character_event = {
	id = 38060
	desc = EVTDESC38060
	picture = GFX_evt_council

	only_playable = yes
	min_age = 16
	prisoner = no
	only_capable = yes

	trigger = {
		wealth >= 250
		independent = no
		NOT = { has_character_flag = offered_to_buy_liege_titles }

		liege = {
			wealth < -50

			any_demesne_title = {
				lower_real_tier_than = PREV
				has_de_jure_pretension = ROOT
				can_be_given_away = yes

				trigger_if = {
					limit = { tier = COUNT }

					location = {
						is_capital = no
					}
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 60

		mult_modifier = {
			factor = 0.75
			liege = { wealth < -75 }
		}

		mult_modifier = {
			factor = 0.75
			liege = { wealth < -100 }
		}

		mult_modifier = {
			factor = 0.75
			liege = { wealth < -125 }
		}

		mult_modifier = {
			factor = 0.75
			liege = { wealth < -150 }
		}

		mult_modifier = {
			factor = 0.75
			wealth >= 275
		}

		mult_modifier = {
			factor = 0.75
			wealth >= 300
		}

		mult_modifier = {
			factor = 0.75
			wealth >= 325
		}

		mult_modifier = {
			factor = 0.75
			wealth >= 350
		}
	}

	immediate = {
		set_character_flag = offered_to_buy_liege_titles

		liege = {
			random_demesne_title = {
				limit = {
					lower_real_tier_than = PREV
					has_de_jure_pretension = ROOT
					can_be_given_away = yes

					trigger_if = {
						limit = { tier = COUNT }

						location = {
							is_capital = no
						}
					}
				}

				save_event_target_as = title_for_sale
			}
		}
	}

	option = {
		name = EVTOPTC38060

		trigger = {
			has_dlc = "Conclave"

			liege = {
				NOT = { owes_favor_to = ROOT }
			}
		}

		ai_chance = { factor = 100 }

		liege = {
			show_scope_change = no

			character_event = {
				id = 38062
				days = 1
				tooltip = EVTTOOLTIP38061
			}
		}

		tooltip = {
			grant_title = event_target:title_for_sale
			wealth = -300
			add_favor = liege
		}
	}

	option = {
		name = EVTOPTA38060

		ai_chance = { factor = 100 }

		liege = {
			show_scope_change = no

			character_event = {
				id = 38061
				days = 1
				tooltip = EVTTOOLTIP38061
			}
		}

		tooltip = {
			grant_title = event_target:title_for_sale
			wealth = -200
		}
	}

	option = {
		name = EVTOPTB38060

		ai_chance = { factor = 0 }
	}
}
