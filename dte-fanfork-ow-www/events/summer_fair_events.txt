character_event = {
	id = 74000
	desc = EVTDESC74000
	picture = GFX_evt_jugglers

	only_rulers = yes
	min_age = 20
	only_capable = yes
	prisoner = no

	trigger = {
		has_character_modifier = holding_summer_fair

		had_character_flag = {
			flag = summer_fair_begin
			days >= 3
		}

		trigger_if = {
			limit = { has_character_flag = fair_event_happened }

			had_character_flag = {
				flag = fair_event_happened
				days >= 3
			}
		}

		capital_scope = {
			NOT = { has_province_modifier = peasants_upset }
		}

		NOT = { has_character_flag = fair_event_74000 }
	}

	immediate = {
		set_character_flag = fair_event_happened
		set_character_flag = fair_event_74000
	}

	mean_time_to_happen = {
		days = 180

		mult_modifier = {
			factor = 0.33
			NOT = { has_character_flag = fair_event_happened }
		}

		mult_modifier = {
			factor = 1.5
			has_character_flag = fair_event_happened
		}

		mult_modifier = {
			factor = 0.5
			trait = charitable
		}

		mult_modifier = {
			factor = 2
			trait = greedy
		}
	}


	option = {
		name = EVTOPTA74000

		trigger = {
			NOR = {
				trait = kind
				trait = charitable
				trait = honest
			}
		}

		random = {
			chance = 10
			add_trait_stressed_effect = yes
		}
	}

	option = {
		name = EVTOPTB74000

		trigger = {
			OR = {
				trait = kind
				trait = charitable
				trait = honest
			}

			any_vassal = {
				OR = {
					trait = kind
					trait = charitable
					trait = honest
				}

				is_adult = yes
				prisoner = no
				is_incapable = no
				is_inaccessible_trigger = no
			}
		}

		random_vassal = {
			show_scope_change = no

			limit = {
				OR = {
					trait = kind
					trait = charitable
					trait = honest
				}

				is_adult = yes
				prisoner = no
				is_incapable = no
				is_inaccessible_trigger = no
			}

			opinion = {
				name = honest_pos
				who = ROOT
				years = 5
			}
		}
	}

	option = {
		name = EVTOPTC74000

		trigger = { trait = arbitrary }

		capital_scope = {
			show_scope_change = no

			add_province_modifier = {
				name = peasants_upset
